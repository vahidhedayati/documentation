                print code1;
        }
        if( match($0, /code=.*"[a-z]+"/) ) {
                str1=substr($0, RSTART, RLENGTH);
                code=substr(str1, index(str1,"="));
                code1=substr(code,  index(code,"=")+index(code," "),index(code,"\"")-index(code,"\""));
                print code1;
        }

   }'|sort|uniq
